
min_activity_age = {
	value = 16
	if = {
		limit = { has_game_rule = min_activity_age_eight }
		value = 8
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_nine }
		value = 9
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_ten }
		value = 10
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_eleven }
		value = 11
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_twelve }
		value = 12
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_thirteen }
		value = 13
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_fourteen }
		value = 14
	}
	else_if = {
		limit = { has_game_rule = min_activity_age_fifteen }
		value = 15
	}
}

# become adventurer
become_adventurer_gold_cost = {
	value = 0
	if = {
		limit = {
			has_game_rule = adventurer_cultural_unlock_relaxed
			become_adventurer_default_cultural_trigger = no
		}
		add = 40
		add = {
			value = highest_held_title_tier
			multiply = 10
		}
	}
	if = {
		limit = {
			has_game_rule = adventurer_martial_unlock_relaxed
			save_temporary_scope_as = potential_laamp_temp
			become_adventurer_default_martial_trigger = no
		}
		add = 40
		add = {
			value = highest_held_title_tier
			multiply = 10
		}
	}
	if = {
		limit = {
			has_game_rule = adventurer_roaming_unlock_relaxed
			become_adventurer_default_roaming_trigger = no
		}
		add = 40
		add = {
			value = highest_held_title_tier
			multiply = 10
		}
	}
}
