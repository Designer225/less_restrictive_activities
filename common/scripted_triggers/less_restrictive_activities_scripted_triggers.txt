## activity-specific
# inspection
can_start_inspection = {
	switch = {
		trigger = has_game_rule
		inspection_unlock_perk = {
			OR = {
				has_perk = local_inspection_perk
				government_allows = administrative
				faith = { has_doctrine_parameter = inspections_unlocked_and_costs_piety }
			}
		}
		inspection_unlock_perk_or_child = {
			OR = {
				has_perk = local_inspection_perk
				government_allows = administrative
				faith = { has_doctrine_parameter = inspections_unlocked_and_costs_piety }
				is_adult = no
			}
		}
		inspection_unlock_focus = {
			OR = {
				has_perk = local_inspection_perk
				government_allows = administrative
				faith = { has_doctrine_parameter = inspections_unlocked_and_costs_piety }
				has_focus = wanderer_internal_affairs_focus
			}
		}
		inspection_unlock_focus_or_child = {
			OR = {
				has_perk = local_inspection_perk
				government_allows = administrative
				faith = { has_doctrine_parameter = inspections_unlocked_and_costs_piety }
				has_focus = wanderer_internal_affairs_focus
				is_adult = no
			}
		}
		inspection_unlock_none = {
			always = yes
		}
	}
}

# become an adventurer
become_adventurer_default_cultural_trigger = {
	OR = {
		culture = { has_cultural_parameter = unlock_voluntary_laampdom }
		faith = { has_doctrine_parameter = unlock_voluntary_laampdom_faith }
		has_trait = nomadic_philosophy
		has_character_modifier = tgp_gave_up_modifier
		AND = {
			is_ai = no
			government_allows = administrative
			top_liege = this
			is_landed = no
		}
	}
}

become_adventurer_cultural_trigger = {
	switch = {
		trigger = has_game_rule
		adventurer_cultural_unlock_default = {
			become_adventurer_default_cultural_trigger = yes
		}
		adventurer_cultural_unlock_relaxed = {
			always = yes # unlock governed by price instead, with the same trigger
		}
		adventurer_cultural_unlock_player_only = {
			OR = {
				become_adventurer_default_cultural_trigger = yes
				is_ai = no
			}
		}
		adventurer_cultural_unlock_free = {
			always = yes
		}
	}
}
